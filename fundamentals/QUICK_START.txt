from valuation import DCFValuation, ComparableValuation

# DCF Model
dcf = DCFValuation(wacc=0.08, terminal_growth_rate=0.03)
value_per_share, assumptions = dcf.calculate_enterprise_value(
    base_fcf=1000000,
    growth_rates=[0.15, 0.12, 0.10, 0.08, 0.05],
    shares_outstanding=1000000,
    net_debt=5000000
)

# Sensitivity Analysis
sensitivity = dcf.sensitivity_analysis(
    base_fcf=1000000,
    growth_rates=[0.15, 0.12, 0.10, 0.08, 0.05],
    shares_outstanding=1000000,
    net_debt=5000000,
    wacc_range=(0.06, 0.10),
    tgr_range=(0.02, 0.04)
)

# Comparable Valuation
comparable_vals = ComparableValuation.calculate_metrics_from_comparables(
    comparable_pe=[25.0, 28.5, 26.3],
    comparable_pb=[3.2, 3.5, 3.1],
    comparable_ps=[5.1, 5.8, 5.3],
    target_eps=5.50,

PEER COMPARISON
───────────────
from peer_comparison import PeerComparator, PeerMetrics, ComparisonMetric

comparator = PeerComparator()

# Add peers
peer = PeerMetrics(
    ticker='MSFT',
    company_name='Microsoft',
    sector='Technology',
    industry='Software',
    market_cap=3.0e12,
    metrics={
        ComparisonMetric.PE_RATIO: 32.1,
        ComparisonMetric.ROE: 42.5,
        ComparisonMetric.DEBT_TO_EQUITY: 0.3
    },
    growth_rate=0.12,
    quality_score=92.0
)
comparator.add_peer(peer)

# Get comprehensive comparison
result = comparator.comprehensive_comparison('AAPL', limit=10)
print(f"Percentile Ranks: {result.percentile_rank}")
print(f"Recommendation: {result.recommendation}")
